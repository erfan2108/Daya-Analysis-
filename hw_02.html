۲. نمودار زمانی طول عرض ضخامت و کیفیت دوربین گوشی ها را بکشید و به نمودار هر یک مدل خطی اضافه کنید. 

# tul
mobile %>% group_by(year) %>% summarise(dimLength = mean(dim_length, na.rm = T)) %>% na.omit() -> len
p = ggplot(data = len, aes(x = year, y= dimLength))
coef = coef(lm(dimLength ~ year, data = len))
p + geom_point() + geom_abline(intercept = coef[1], slope=coef[2]) + ggtitle('Length VS year')

# arz

mobile %>% select(company, device, year, cam_px, sel_px) %>% na.omit() %>% group_by(year) %>% summarise(ratio = mean(sel_px) / mean(cam_px)) -> mod

این نمودار تا سال ۲۰۱۲ مقدار ثابتی دارد اما با معروف شدن اینستاگرام و دیگر شبکه های اجتماعی در سال ۲۰۱۲ رشد نمودار کاملا نمایی می شود

نمودار بعدی نمودار میانگین چگالی گوشی ها بر اساس زمان است که نکته جالب آن صعودی بودن آن است یعنی با تکنولوژی موبایل گوشی ها چگال تر می شوند 

mod = mobile[c('year','dim_length','dim_thickness','dim_breadth','weight')] %>% na.omit()
mod = mod[which(mod$dim_length * mod$dim_breadth * mod$dim_thickness != 0), ]
mod$density = mod$weight/(mod$dim_length * mod$dim_breadth * mod$dim_thickness/1000)
mod %>%  group_by(year) %>% summarise(denMean = mean(density)) -> mod
p = ggplot(data = mod, aes(x=year, y = denMean))
p + geom_point() + ggtitle('Density VS Year')
